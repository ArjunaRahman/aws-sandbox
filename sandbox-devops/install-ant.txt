INSTALL ANT

Move the desired Ant - apache-ant-1.10.2-bin.tar.gz to the desired folder*;

	apache-ant-1.10.2-bin.tar.gz  

cd desired folder*/
sudo chmod a+x apache-ant-1.10.2-bin.tar.gz

sudo tar -xvzf apache-ant-1.10.2-bin.tar.gz -C /dscdev/apps/ant --strip-components=1

sudo nano /etc/profile.d/ant.sh

export ANT_HOME=/dscdev/apps/ant
export PATH=${ANT_HOME}/bin:${PATH}

source /etc/profile
ant -version
